# 校园二手交易平台 - 数据库表结构（适配当前前端功能）

> 本文档对应当前已完成的前端功能（普通用户 + 管理员），使用 MySQL DDL 描述数据表结构，便于后续直接建库建表。
> 
> - 字符集统一使用 `utf8mb4`，存储引擎统一 `InnoDB`。
> - 所有时间字段使用 `datetime`，默认 `CURRENT_TIMESTAMP`，并在需要时开启 `ON UPDATE`。
> - 下方 SQL 可以整体复制执行：**先建库，再建表**。

---

```sql
-- ============================
-- 0. 创建数据库
-- ============================
CREATE DATABASE IF NOT EXISTS `campus_secondhand`
  DEFAULT CHARACTER SET utf8mb4
  DEFAULT COLLATE utf8mb4_general_ci;

USE `campus_secondhand`;

-- ============================
-- 1. 用户表 (user)
-- ============================
CREATE TABLE `user` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '用户ID (主键)',
  `username`    VARCHAR(50)  NOT NULL COMMENT '用户名 (学号/手机号/邮箱)',
  `password`    VARCHAR(100) NOT NULL COMMENT '密码 (加密存储)',
  `nickname`    VARCHAR(50)  NOT NULL DEFAULT '' COMMENT '昵称',
  `avatar`      VARCHAR(255) NOT NULL DEFAULT '' COMMENT '头像 URL',
  `phone`       VARCHAR(20)  DEFAULT NULL COMMENT '手机号码',
  `email`       VARCHAR(50)  DEFAULT NULL COMMENT '电子邮箱',
  `gender`      VARCHAR(10)  NOT NULL DEFAULT '保密' COMMENT '性别 (男/女/保密)',
  `bio`         VARCHAR(255) NOT NULL DEFAULT '' COMMENT '个人简介',
  `status`      TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '状态 (0 正常,1 禁用)',
  `role`        TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '角色 (0 普通用户,1 管理员)',
  `is_certified` TINYINT(1)  NOT NULL DEFAULT 0 COMMENT '校园认证 (0 否,1 是)',
  `balance`     DECIMAL(10,2) NOT NULL DEFAULT 0.00 COMMENT '账户余额',
  `last_read_notification_id` BIGINT(20) DEFAULT 0 COMMENT '最后阅读的系统通知ID',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_username` (`username`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

-- ============================
-- 2. 用户默认收货信息表 (user_address)
--    目前前端只用一条默认地址，可不做多地址管理
-- ============================
CREATE TABLE `user_address` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '地址ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '用户ID',
  `name`        VARCHAR(50) NOT NULL COMMENT '收货人姓名',
  `phone`       VARCHAR(20) NOT NULL COMMENT '收货人电话',
  `address`     VARCHAR(255) NOT NULL COMMENT '详细地址',
  `is_default`  TINYINT(1) NOT NULL DEFAULT 1 COMMENT '是否默认地址 (目前可全为1)',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户默认收货地址表';

-- ============================
-- 3. 商品表 (product)
-- ============================
CREATE TABLE `product` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '商品ID',
  `seller_id`   BIGINT(20) NOT NULL COMMENT '卖家ID',
  `category_id` INT(11)    DEFAULT NULL COMMENT '分类ID (可选, 目前前端未使用分类管理)',
  `title`       VARCHAR(100) NOT NULL COMMENT '商品标题',
  `description` TEXT COMMENT '商品描述',
  `price`       DECIMAL(10,2) NOT NULL COMMENT '售价',
  `condition`   VARCHAR(20)  DEFAULT NULL COMMENT '新旧程度 (全新/九成新/...)',
  `status`      TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '状态 (0:在售, 1:已售出, 2:已下架)',
  `views`       INT(11)      NOT NULL DEFAULT 0 COMMENT '浏览次数',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_seller_id` (`seller_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品表';

-- ============================
-- 4. 商品图片表 (product_image)
-- ============================
CREATE TABLE `product_image` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '图片ID',
  `product_id`  BIGINT(20) NOT NULL COMMENT '商品ID',
  `image_url`   VARCHAR(255) NOT NULL COMMENT '图片URL',
  `is_primary`  TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否主图',
  `sort_order`  INT(11)    NOT NULL DEFAULT 0 COMMENT '排序',
  `create_time` DATETIME   NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_product_id` (`product_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品图片表';

-- ============================
-- 5. 商品草稿表 (product_draft)
-- ============================
CREATE TABLE `product_draft` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '草稿ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '用户ID',
  `title`       VARCHAR(100) NOT NULL COMMENT '商品标题(草稿)',
  `description` TEXT COMMENT '商品描述(草稿)',
  `price`       DECIMAL(10,2) DEFAULT NULL COMMENT '价格(草稿)',
  `condition`   VARCHAR(20)   DEFAULT NULL COMMENT '新旧程度(草稿)',
  `images_json` TEXT COMMENT '图片URL数组(JSON)',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品草稿表';

-- ============================
-- 6. 收藏表 (favorite)
-- ============================
CREATE TABLE `favorite` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '收藏ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '用户ID',
  `product_id`  BIGINT(20) NOT NULL COMMENT '商品ID',
  `status`      TINYINT(1) NOT NULL DEFAULT 1 COMMENT '状态 (1 有效,2 失效)',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_user_product` (`user_id`, `product_id`),
  KEY `idx_product_id` (`product_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='收藏表';

-- ============================
-- 7. 订单表 (orders)
-- ============================
-- 状态: 0待付款,1待发货,2待收货,3待评价,4已完成,5退款中,6已取消,7退款成功,8退款失败
CREATE TABLE `orders` (
  `id`            BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '订单ID',
  `order_no`      VARCHAR(50) NOT NULL COMMENT '订单号',
  `product_id`    BIGINT(20) NOT NULL COMMENT '商品ID',
  `buyer_id`      BIGINT(20) NOT NULL COMMENT '买家ID',
  `seller_id`     BIGINT(20) NOT NULL COMMENT '卖家ID',
  `order_price`   DECIMAL(10,2) NOT NULL COMMENT '成交价格',
  `status`        TINYINT(1)   NOT NULL DEFAULT 0 COMMENT '订单状态',
  `shipping_name` VARCHAR(50)  NOT NULL COMMENT '收货人姓名',
  `shipping_phone` VARCHAR(20) NOT NULL COMMENT '收货人电话',
  `shipping_address` VARCHAR(255) NOT NULL COMMENT '收货地址',
  `create_time`   DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `pay_time`      DATETIME DEFAULT NULL COMMENT '付款时间',
  `ship_time`     DATETIME DEFAULT NULL COMMENT '发货时间',
  `complete_time` DATETIME DEFAULT NULL COMMENT '成交时间',
  `cancel_time`   DATETIME DEFAULT NULL COMMENT '取消时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_order_no` (`order_no`),
  KEY `idx_buyer_id` (`buyer_id`),
  KEY `idx_seller_id` (`seller_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单表';

-- ============================
-- 8. 退款主表 (order_refund)
-- ============================
CREATE TABLE `order_refund` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '退款记录ID',
  `order_id`    BIGINT(20) NOT NULL COMMENT '订单ID',
  `status`      TINYINT(1) NOT NULL DEFAULT 0 COMMENT '退款状态 (0 退款中,1 退款成功,2 退款失败)',
  `reason`      VARCHAR(100) DEFAULT NULL COMMENT '买家选择的退款原因',
  `description` VARCHAR(255) DEFAULT NULL COMMENT '买家退款说明',
  `dispute_in_progress` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否发起维权',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_order_id` (`order_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单退款主表';

-- ============================
-- 9. 退款历史表 (order_refund_history)
-- ============================
CREATE TABLE `order_refund_history` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '历史ID',
  `refund_id`   BIGINT(20) NOT NULL COMMENT '退款ID',
  `text`        VARCHAR(255) NOT NULL COMMENT '进度描述文案',
  `is_rejection` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否为卖家拒绝记录',
  `reason`      VARCHAR(100) DEFAULT NULL COMMENT '拒绝/退款原因',
  `description` VARCHAR(255) DEFAULT NULL COMMENT '说明',
  `rejection_timestamp` BIGINT(20) DEFAULT NULL COMMENT '卖家拒绝时间戳(用于72小时倒计时)',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录时间',
  PRIMARY KEY (`id`),
  KEY `idx_refund_id` (`refund_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单退款进度历史表';

-- ============================
-- 10. 订单评价表 (order_review)
-- ============================
CREATE TABLE `order_review` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '评价ID',
  `order_id`    BIGINT(20) NOT NULL COMMENT '订单ID',
  `reviewer_id` BIGINT(20) NOT NULL COMMENT '评价人ID',
  `target_type` VARCHAR(10) NOT NULL COMMENT '评价对象 (buyer/seller)',
  `rating`      TINYINT(1)  DEFAULT NULL COMMENT '评分 (买家评价卖家时使用 1-5)',
  `content`     VARCHAR(500) DEFAULT '' COMMENT '评价内容',
  `is_additional` TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否追评 (0 否,1 是)',
  `parent_id`   BIGINT(20) DEFAULT NULL COMMENT '关联的父评价ID (用于卖家回复)',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_order_id` (`order_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='订单评价表';

-- ============================
-- 11. 商品留言表 (product_comment)
-- ============================
CREATE TABLE `product_comment` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '留言ID',
  `product_id`  BIGINT(20) NOT NULL COMMENT '商品ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '留言用户ID',
  `content`     VARCHAR(500) NOT NULL COMMENT '留言内容',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_product_id` (`product_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品留言表';

-- ============================
-- 12. 商品留言回复表 (product_comment_reply)
-- ============================
CREATE TABLE `product_comment_reply` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '回复ID',
  `comment_id`  BIGINT(20) NOT NULL COMMENT '所属留言ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '回复用户ID',
  `content`     VARCHAR(500) NOT NULL COMMENT '回复内容',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_comment_id` (`comment_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='商品留言回复表';

-- ============================
-- 13. 私信会话表 (conversation)
-- ============================
CREATE TABLE `conversation` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '会话ID',
  `user1_id`    BIGINT(20) NOT NULL COMMENT '用户1 ID',
  `user2_id`    BIGINT(20) NOT NULL COMMENT '用户2 ID',
  `is_support`  TINYINT(1) NOT NULL DEFAULT 0 COMMENT '是否为客服/维权会话',
  `last_message` VARCHAR(255) DEFAULT NULL COMMENT '最后一条消息概要',
  `last_time`   DATETIME DEFAULT NULL COMMENT '最后消息时间',
  PRIMARY KEY (`id`),
  KEY `idx_user1_id` (`user1_id`),
  KEY `idx_user2_id` (`user2_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='私信会话表';

-- ============================
-- 14. 私信消息表 (conversation_message)
-- ============================
CREATE TABLE `conversation_message` (
  `id`             BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '消息ID',
  `conversation_id` BIGINT(20) NOT NULL COMMENT '会话ID',
  `sender_id`      BIGINT(20) NOT NULL COMMENT '发送方ID',
  `content`        TEXT       NOT NULL COMMENT '消息内容',
  `is_read`        TINYINT(1) NOT NULL DEFAULT 0 COMMENT '已读状态',
  `create_time`    DATETIME   NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  KEY `idx_conversation_id` (`conversation_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='私信消息表';

-- ============================
-- 15. 系统通知表 (notification)
-- ============================
CREATE TABLE `notification` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '通知ID',
  `title`       VARCHAR(100) NOT NULL COMMENT '通知标题',
  `content`     VARCHAR(500) NOT NULL COMMENT '通知内容',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='系统通知表';

-- ============================
-- 16. 钱包交易流水表 (wallet_transaction)
-- ============================
CREATE TABLE `wallet_transaction` (
  `id`          BIGINT(20) NOT NULL AUTO_INCREMENT COMMENT '记录ID',
  `user_id`     BIGINT(20) NOT NULL COMMENT '用户ID',
  `type`        TINYINT(1) NOT NULL COMMENT '类型 (0:充值, 1:消费, 2:提现, 3:卖出收入)',
  `amount`      DECIMAL(10,2) NOT NULL COMMENT '变动金额',
  `description` VARCHAR(255) NOT NULL COMMENT '描述文案',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录时间',
  PRIMARY KEY (`id`),
  KEY `idx_user_id` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='钱包交易流水表';
```
